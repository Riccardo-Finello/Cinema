CREATE TABLE `cinema`.`film` ( `ID` INT(8) NOT NULL AUTO_INCREMENT ,
 `Title` VARCHAR(100) NOT NULL ,
 `Plot` VARCHAR(1500) NOT NULL ,
 `Thumbnail` VARCHAR(200) NOT NULL ,
 PRIMARY KEY (`ID`)) 
 ENGINE = InnoDB;

 CREATE TABLE `proiezioni` ( `ID` INT(8) NOT NULL AUTO_INCREMENT ,
 `Movie` INT(8) NOT NULL ,
 `Date` DATETIME NOT NULL ,
 `Room` INT(3) NOT NULL, 
 PRIMARY KEY (`ID`)) 
 ENGINE = InnoDB;
